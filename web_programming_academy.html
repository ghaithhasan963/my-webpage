JavaScript - قوة التفاعل

JavaScript هي لغة البرمجة التي تجعل مواقع الويب تفاعلية وديناميكية. تعلم كيفية إضافة السلوك والتفاعل لمواقعك. 

// تغيير النص عند الضغط على الزر
function changeText() {
    const element = document.getElementById('myText');
    element.textContent = 'النص تم تغييره!';
    element.style.color = '#667eea';
}

// إضافة مستمع الأحداث
document.getElementById('myButton').addEventListener('click', changeText);

// التحقق من صحة النموذج
function validateForm() {
    const email = document.getElementById('email').value;
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    
    if (!emailRegex.test(email)) {
        alert('الرجاء إدخال بريد إلكتروني صحيح');
        return false;
    }
    return true;
}

React - واجهات المستخدم الحديثة

React هي مكتبة JavaScript قوية لبناء واجهات المستخدم التفاعلية. تعلم كيفية إنشاء تطبيقات ويب حديثة وسريعة. 

import React, { useState } from 'react';

function Counter() {
    const [count, setCount] = useState(0);

    return (
        <div className="counter">
            <h2>العداد: {count}</h2>
            <button onClick={() => setCount(count + 1)}>
                زيادة
            </button>
            <button onClick={() => setCount(count - 1)}>
                تقليل
            </button>
        </div>
    );
}

export default Counter;

ما ستتعلمه:

  • المكونات والـ Props
  • إدارة الحالة

Vue.js - التطوير التقدمي

Vue.js هو إطار عمل JavaScript تقدمي وسهل التعلم. تعلم كيفية بناء تطبيقات تفاعلية بطريقة بسيطة وفعالة. 

<template>
  <div class="todo-app">
    <h1>قائمة المهام</h1>
    <input v-model="newTodo" @keyup.enter="addTodo" placeholder="أضف مهمة جديدة">
    <ul>
      <li v-for="todo in todos" :key="todo.id">
        {{ todo.text }}
        <button @click="removeTodo(todo.id)">حذف</button>
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  data() {
    return {
      newTodo: '',
      todos: []
    }
  },
  methods: {
    addTodo() {
      if (this.newTodo.trim()) {
        this.todos.push({
          id: Date.now(),
          text: this.newTodo
        });
        this.newTodo = '';
      }
    },
    removeTodo(id) {
      this.todos = this.todos.filter(todo => todo.id !== id);
    }
  }
}
</script>

ما ستتعلمه:

  • التوجيهات والبيانات المربوطة
  • إدارة الحالة مع Vuex
  • التوجيه مع Vue Router
  • التطبيقات أحادية الصفحة
